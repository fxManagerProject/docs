## Ban exports

### `recentBans`

Retrieves a list of recent bans.

**Parameters:**
<TypeTable
  type={{
    options: {
      type: 'PageSize',
      required: false,
      description: (
        <div>
          An optional configuration object for configuring pagination:
          <ul className="mt-2 list-disc pl-4 text-xs space-y-1">
            <li><code>page</code>: Target page offset.</li>
            <li><code>pageSize</code>: Total items returned per frame.</li>
          </ul>
        </div>
      ),
    },
  }}
/>

<Accordions type="single">
  <Accordion title="Response Format">
      ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
          id: number;
          playerId: number;
          name: string;
          reason: string;
          issuer: number | null;
          createdAt: Date;
          expiresAt: Date | null;
          revokedAt: Date | null;
        }[];
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
local options = {
  page = 1;
  pageSize = 5;
}
local result = exports.fxmanager:recentBans(options)

if result.ok then
    for _, ban in ipairs(result.data) do
        print("Bane (".. ban.id ..") Player: " .. ban.name .. " - " .. ban.reason)
    end
else
    print("Failed to fetch bans: " .. result.error)
end

```

```js tab="JS"
const options = {
  page: 1,
  pageSize: 5,
};
const result = await exports.fxmanager:recentBans(options);

if (result.ok) {
  result.data.forEach(({ id, name, reason }) =>
    console.log(`Ban: (${id}) Player: ${name} - ${reason}`)
  );
} else {
  console.error(`Failed to fetch bans: ${result.error}`);
}
```

### `ban`

**Payload:**
<TypeTable
  type={{
    target: {
      type: 'Target',
      required: true,
      description: (
        <div>
          <p>One of the following target types:</p>
          <ul className="mt-2 list-disc pl-4 text-xs space-y-1">
            <li><code>number</code>: The player's server ID.</li>
            <li><code>{'{ serverId: number }'}</code>: The player's server ID.</li>
            <li><code>{'{ playerId: number }'}</code>: The player's database ID.</li>
            <li><code>{'{ identifiers: { license: string } }'}</code>: Direct lookup using Rockstar license hashes.</li>
          </ul>
        </div>
      ),
    },
    reason: {
      type: 'string',
      required: true,
      description: 'The reason or enforcement context attached to the ban.',
    },
    expiresAt: {
      type: 'string | null',
      required: false,
      description: 'An ISO timestamp string detailing when the ban lifts. Pass null for a permanent ban.',
    },
    durationSeconds: {
      type: 'number | null',
      required: false,
      description: 'Alternative way to specify ban length in seconds. Pass null for a permanent ban.',
    },
    by: {
      type: 'number',
      required: true,
      description: 'The database ID of the Admin/Staff member issuing the ban.',
    },
  }}
/>

<Accordions type="single">
  <Accordion title="Response Format">
      ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
          banId: number;
        };
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
local target = { serverId = 1 }
local reason = "No reason provided"
local expiresAt = nil
local durationSeconds = nil
local by = 1

local response = exports.fxManager:ban(target, reason, expiresAt, durationSeconds, by)

if response.ok then
  print("Ban successful, banId:", response.data.banId)
else
  print("Unable to ban:", response.error)
end
```
```ts tab="JS"
const target = { serverId: 1 };
const reason = "No reason provided";
const expiresAt = null;
const durationSeconds = null;
const by = 1;

const response = await exports.fxManager.ban(target, reason, expiresAt, durationSeconds, by);
if (response.ok) {
  console.log("Ban successful, banId:", response.data.banId);
} else {
  console.log("Unable to ban:", response.error);
}
```

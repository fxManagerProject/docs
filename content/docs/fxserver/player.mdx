Methods for retrieving live connection data, searching historical player files, or querying specific player targets.

### `fetchPlayers`

Retrieves an array of all active players currently initialized on the game server.

<Accordions type="single">
  <Accordion title="Response Format">
      ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
         	id: number;
         	name: string;
         	playtime: number;
         	identifiers: Record<string, string>;
         	isStaff: boolean;
         	firstSeen: Date;
         	lastSeen: Date;
         	serverId: number;
         	ping?: number;
         	health: number;
        }[];
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
local result = exports.fxmanager:fetchPlayers()

if result.ok then
    for _, player in ipairs(result.data) do
        print("Active Player: " .. player.name)
    end
else
    print("Failed to fetch players: " .. result.error)
end

```

```js tab="JS"
if (result.ok) {
  result.data.forEach(player =>
    console.log(`Active Player: ${player.name}`)
  );
} else {
  console.error(`Failed to fetch players: ${result.error}`);
}
```

### `getPlayer`
Looks up an individual player record based on a structured `Target` descriptor. 

**Parameters:**
<TypeTable
  type={{
    target: {
      type: 'Target',
      required: true,
      description: (
        <div>
          <p>One of the following target types:</p>
          <ul className="mt-2 list-disc pl-4 text-xs space-y-1">
            <li><code>number</code>: The player's server ID.</li>
            <li><code>{'{ serverId: number }'}</code>: The player's server ID.</li>
            <li><code>{'{ playerId: number }'}</code>: The player's database ID.</li>
            <li><code>{'{ identifiers: { license: string } }'}</code>: Direct lookup using Rockstar license hashes.</li>
          </ul>
        </div>
      ),
    },
  }}
/>

<Accordions type="single">
  <Accordion title="Response Format">
      ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
         	id: number;
         	name: string;
         	playtime: number;
         	identifiers: Record<string, string>;
         	isStaff: boolean;
         	firstSeen: Date;
         	lastSeen: Date;
         	serverId: number;
         	ping?: number;
         	health: number;
        };
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
-- Lookup via standard Server ID
local target = 14

-- Alternate: Lookup via license string
-- local target = { identifiers = { license = "license:xxxxxxxx" } }

local result = exports.fxmanager:getPlayer(target)

if result.ok then
    print("Found Player: " .. result.data.name .. " | Playtime: " .. result.data.playtime)
else
    print("Lookup failed: " .. result.error)
end

```

```js tab="JS"
if (result.ok) {
  console.log(`Found Player: ${result.data.name}`);
} else {
  console.error(`Lookup failed: ${result.error}`);
}

```

### `getSelf`

Retrieves the administrative profile and permission data for a specific player based on their server ID.

**Parameters:**
<TypeTable
  type={{
    playerId: {
      type: 'number | string',
      required: true,
      description: 'The server ID of the player whose admin information is being requested.',
    },
  }}
/>

<Accordions type="single">
<Accordion title="Response Format">
    ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
            isAdmin: boolean;
           	isMaster: boolean;
           	adminId: number | null;
           	username: string | null;
           	group: { id: number; name: string; colour: string; icon: string | null } | null;
           	permissions: string[];
        }
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
-- The server ID of the player
local playerId = 1

local result = exports.fxmanager:getSelf(playerId)

if result.ok then
    if result.data.isAdmin then
        print("Admin found: " .. tostring(result.data.username) .. " | Is Master: " .. tostring(result.data.isMaster))
    else
        print("Player is not an admin.")
    end
else
    print("Lookup failed: " .. result.error)
end

```

```js tab="JS"
const playerId = 1;
const result = exports.fxmanager.getSelf(playerId);

if (result.ok) {
  if (result.data.isAdmin) {
    console.log(`Admin found: ${result.data.username} | Group: ${result.data.group?.name}`);
  } else {
    console.log("Player is not an admin.");
  }
} else {
  console.error(`Lookup failed: ${result.error}`);
}

```

### `searchPlayers`
Searches the entire historical player database using text match strings alongside pagination and sorting options.

**Parameters:**
<TypeTable
  type={{
    query: {
      type: 'string',
      required: true,
      description: 'The search term (e.g., player name, Rockstar license, Discord ID, or ban reason).',
    },
    options: {
      type: 'Pagination',
      required: false,
      description: (
        <div>
          <p>An optional configuration object to customize sorting and pagination limits:</p>
          <ul className="mt-2 list-disc pl-4 text-xs space-y-1">
            <li>
              page: <code>number</code> - The page number to retrieve for pagination.
            </li>
            <li>
              pageSize: <code>number</code> - The number of results to return per page.
            </li>
            <li>
              sortBy: <code>{"'playtime' | 'lastSeen' | 'firstSeen'"}</code> - The field key used to sort the resulting array.
            </li>
            <li>
              sortOrder: <code>{"'asc' | 'desc'"}</code> - The direction to sort the matches.
            </li>
          </ul>
        </div>
      ),
    },
  }}
/>

<Accordions type="single">
<Accordion title="Response Format">
    ```ts tab="TypeScript"
      type Result = {
        ok: true;
        data: {
         	page: number;
          pageSize: number;
          total: number;
          items: {
            id: number;
            name: string;
            playtime: number;
            isStaff: boolean;
            firstSeen: Date;
            lastSeen: Date;
          }[];
      } | {
        ok: false;
        error: string;
      }
      ```
  </Accordion>
</Accordions>

```lua tab="Lua"
local opts = {
    page = 1,
    pageSize = 10,
    sortBy = 'playtime',
    sortOrder = 'desc'
}

local result = exports.fxmanager:searchPlayers("John", opts)

if result.ok then
    print("Found " .. #result.data .. " matches.")
end
```

```js tab="JS"
const opts = {
  page: 1,
  pageSize: 10,
  sortBy: 'playtime',
  sortOrder: 'desc',
};

const result = await exports.fxmanager.searchPlayers("GamerName", opts);
```

### `addNote`

Appends an administrative note to a player's file history. The invoking resource name is automatically captured in the background for auditing.

**Parameters:**
<TypeTable
  type={{
    target: {
      type: 'Target',
      required: true,
      description: (
        <div>
          <p>One of the following target types:</p>
          <ul className="mt-2 list-disc pl-4 text-xs space-y-1">
            <li><code>number</code>: The player's server ID.</li>
            <li><code>{'{ serverId: number }'}</code>: The player's server ID.</li>
            <li><code>{'{ playerId: number }'}</code>: The player's database ID.</li>
            <li><code>{'{ identifiers: { license: string } }'}</code>: Direct lookup using Rockstar license hashes.</li>
          </ul>
        </div>
      ),
    },
    content: {
      type: 'string',
      required: true,
      description: 'The content of the note.',
    },
    by: {
      type: 'number',
      required: true,
      description: 'The database admin ID of the staff member creating the note.',
    },
  }}
/>

<Accordions type="single">
<Accordion title="Response Format">
```ts
type Result = {
  ok: true;
  data: {
    saved: true;
  };
} | {
  ok: false;
  error: string;
}
```
</Accordion>
</Accordions>

```lua tab="Lua"
local notePayload = {
    target = 42, -- Server ID
    content = "Player was warned for aggressive driving patterns.",
    by = 1       -- Admin Database ID
}

local result = exports.fxmanager:addNote(notePayload)
```
```js tab="JS"
const notePayload = {
    target: 42, // Server ID
    content: "Player was warned for aggressive driving patterns.",
    by: 1, // Admin Database ID
}

const result = await exports.fxmanager.addNote(notePayload)
```
